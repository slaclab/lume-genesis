def plot_field(dat, dgrid):
    ndat = np.angle(dat)
    plt.imshow(ndat, extent = [1000*dgrid*i for i in [-1,1,-1,1]])
    plt.xlabel('x (mm)')
    plt.ylabel('y (mm)')
    plt.show()
